CREATE TABLE Zoologico
(
	CNPJ CHAR(20),
	ENDERECO VARCHAR(50),
	TELEFONE CHAR(17),
	ID_FUNCIONARIO INTEGER,
	ID_CLIENTE INTEGER,
	ID_ANIMAL INTEGER,
	ID_VISITA INTEGER,
	ID_FINANCEIRO INTEGER
);

CREATE TABLE Funcionario
(
	ID_FUNCIONARIO INTEGER,
	CPF CHAR(14),
	NOME VARCHAR(30),
	TELEFONE CHAR(17),
	DATA_NASCIMENTO DATE,
	ENDERECO VARCHAR(50),
	SALARIO NUMBER(5,2)
);

CREATE TABLE Cliente
(
	ID_CLIENTE INTEGER,
	CPF CHAR(14),
	NOME VARCHAR(30),
	TELEFONE CHAR(17),
	DATA_NASCIMENTO DATE,
	ENDERECO VARCHAR(50)
);

CREATE TABLE Animais
(
	ID_ANIMAL INTEGER,
	RACA VARCHAR(15),
	REGISTRO_LEGAL INTEGER,
	SETOR VARCHAR(20),
	PESO NUMBER(4,2),
	NASCIMENTO DATE,
	PAIS_ORIGEM VARCHAR(30),
	DATA_AQUISICAO DATE,
	DESCRICAO VARCHAR (500)
);

CREATE TABLE Visitas
(
	ID_VISITA INTEGER,
	DATA DATE,
	VALOR_GASTO NUMBER(4,2),
	FORMA_PAGAMENTO INTEGER,
	DURACAO CHAR(5),
	AVALIACAO INTEGER,
	OPINIAO VARCHAR(500),
	ID_CLIENTE INTEGER
);

CREATE TABLE Financeiro
(
	ID_FINANCEIRO INTEGER,
	MES CHAR(2),
	LUCRO NUMBER(7,2),
	VISITAS INTEGER,
	ID_GASTOS INTEGER,
	ID_GANHOS INTEGER
);

CREATE TABLE Gastos
(
	ID_GASTOS INTEGER,
	SALARIOS NUMBER(6,2),
	RACAO NUMBER(6,2),
	COMPRA_ANIMAIS NUMBER(6,2),
	UTENSILIOS NUMBER(6,2),
	ALIMENTOS NUMBER(6,2),
	BEBIDAS NUMBER(6,2)
);

CREATE TABLE Ganhos
(
	ID_GANHOS INTEGER,
	INGRESSOS NUMBER(6,2),
	ALIMENTOS NUMBER(6,2),
	BEBIDAS NUMBER(6,2),
	ESTACIONAMENTO NUMBER(6,2)
);
