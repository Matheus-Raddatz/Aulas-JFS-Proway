CREATE TABLE LIDER
(
MATRICULA_LIDER INT,
NOME_LIDER VARCHAR(30),
CONSTRAINT MATRICULA_LIDER_PK PRIMARY KEY (MATRICULA_LIDER)
);
CREATE TABLE PROJETO
(
CODIGO_PROJETO INT,
TITULO VARCHAR(20),
CONCEITO VARCHAR(80),
SITUACAO VARCHAR(50),
CONSTRAINT CODIGO_PROJETO_PK PRIMARY KEY (CODIGO_PROJETO)
);
CREATE TABLE CURSOS
(
CODIGO_CURSO INT,
NOME_CURSO VARCHAR(30),
CONSTRAINT CODIGO_CURSO_PK PRIMARY KEY (CODIGO_CURSO)
);
CREATE TABLE PROFESSOR
(
MATRICULA_PROFESSOR INT,
DATA_ADMISSAO VARCHAR(10),
MATRICULA_LIDER INT,
CONSTRAINT MATRICULA_PROFESSOR_PK PRIMARY KEY (MATRICULA_PROFESSOR),
CONSTRAINTS MATRICULA_LIDER_1_FK FOREIGN KEY (MATRICULA_LIDER) REFERENCES LIDER (MATRICULA_LIDER)
);
CREATE TABLE MATERIAS
(
CODIGO_DISCIPLINA INT,
EMENTA VARCHAR(15),
CONTEUDO VARCHAR(15),
MATRICULA_PROFESSOR INT,
CONSTRAINT CODIGO_DISCIPLINA_PK PRIMARY KEY (CODIGO_DISCIPLINA),
CONSTRAINTS MATRICULA_PROFESSOR_1_FK FOREIGN KEY (MATRICULA_PROFESSOR) REFERENCES PROFESSOR (MATRICULA_PROFESSOR)
);
CREATE TABLE TURMA
(
ANO_SEMESTRE VARCHAR(3),
CODIGO_DISCIPLINA INT,
CODIGO_CURSO INT,
CONSTRAINT ANO_SEMESTRE_PK PRIMARY KEY (ANO_SEMESTRE)
);
CREATE TABLE MINISTRA
(
ANO_SEMESTRE VARCHAR(3),
CODIGO_DISCIPLINA INT,
CODIGO_CURSO INT,
MATRICULA_PROFESSOR INT,
CONSTRAINTS ANO_SEMESTRE_1_FK FOREIGN KEY (ANO_SEMESTRE) REFERENCES TURMA (ANO_SEMESTRE),
CONSTRAINTS CODIGO_DISCIPLINA_1_FK FOREIGN KEY (CODIGO_DISCIPLINA) REFERENCES MATERIAS (CODIGO_DISCIPLINA),
CONSTRAINTS CODIGO_CURSO_1_FK FOREIGN KEY (CODIGO_CURSO) REFERENCES CURSOS (CODIGO_CURSO),
CONSTRAINTS MATRICULA_PROFESSOR_2_FK FOREIGN KEY (MATRICULA_PROFESSOR) REFERENCES PROFESSOR (MATRICULA_PROFESSOR)
);
CREATE TABLE ESTUDANTE
(
MATRICULA_ESTUDANTE INT,
NOTA_VESTIBULAR NUMBER(5,2),
CODIGO_CURSO INT,
CONSTRAINT MATRICULA_ESTUDANTE_PK PRIMARY KEY (MATRICULA_ESTUDANTE),
CONSTRAINTS CODIGO_CURSO_2_FK FOREIGN KEY (CODIGO_CURSO) REFERENCES CURSOS (CODIGO_CURSO)
);
CREATE TABLE ESTUDANTE_TURMA
(
ANO_SEMESTRE VARCHAR(3),
CODIGO_DISCIPLINA INT,
CODIGO_CURSO INT,
MATRICULA_ESTUDANTE INT,
CODIGO_PROJETO INT,
CONSTRAINTS ANO_SEMESTRE_2_FK FOREIGN KEY (ANO_SEMESTRE) REFERENCES TURMA (ANO_SEMESTRE),
CONSTRAINTS CODIGO_DISCIPLINA_2_FK FOREIGN KEY (CODIGO_DISCIPLINA) REFERENCES MATERIAS (CODIGO_DISCIPLINA),
CONSTRAINTS CODIGO_CURSO_3_FK FOREIGN KEY (CODIGO_CURSO) REFERENCES CURSOS (CODIGO_CURSO),
CONSTRAINTS MATRICULA_ESTUDANTE_1_FK FOREIGN KEY (MATRICULA_ESTUDANTE) REFERENCES ESTUDANTE (MATRICULA_ESTUDANTE),
CONSTRAINTS CODIGO_PROJETO_1_FK FOREIGN KEY (CODIGO_PROJETO) REFERENCES PROJETO (CODIGO_PROJETO)
);
CREATE TABLE MONITORIA
(
ANO_SEMESTRE VARCHAR(3),
CODIGO_DISCIPLINA INT,
CODIGO_CURSO INT,
MATRICULA_ESTUDANTE INT,
MATRICULA_PROFESSOR INT,
CONSTRAINTS ANO_SEMESTRE_3_FK FOREIGN KEY (ANO_SEMESTRE) REFERENCES TURMA (ANO_SEMESTRE),
CONSTRAINTS CODIGO_DISCIPLINA_3_FK FOREIGN KEY (CODIGO_DISCIPLINA) REFERENCES MATERIAS (CODIGO_DISCIPLINA),
CONSTRAINTS CODIGO_CURSO_4_FK FOREIGN KEY (CODIGO_CURSO) REFERENCES CURSOS (CODIGO_CURSO),
CONSTRAINTS MATRICULA_ESTUDANTE_2_FK FOREIGN KEY (MATRICULA_ESTUDANTE) REFERENCES ESTUDANTE (MATRICULA_ESTUDANTE),
CONSTRAINTS MATRICULA_PROFESSOR_3_FK FOREIGN KEY (MATRICULA_PROFESSOR) REFERENCES PROFESSOR (MATRICULA_PROFESSOR)
);
CREATE TABLE PESSOA
(
MATRICULA_PESSOA INT,
SEXO CHAR(1),
NOME_PESSOA VARCHAR(30),
CONSTRAINT MATRICULA_PESSOA_PK PRIMARY KEY (MATRICULA_PESSOA)
);
CREATE TABLE PROVA
(
DESCRICAO VARCHAR(120),
NOTA DECIMAL(5,2),
CODIGO_DISCIPLINA INT,
CODIGO_CURSO INT,
ANO_SEMESTRE VARCHAR(3),
MATRICULA_ESTUDANTE INT,
CONSTRAINT DESCRICAO_PK PRIMARY KEY (DESCRICAO),
CONSTRAINTS CODIGO_DISCIPLINA_4_FK FOREIGN KEY (CODIGO_DISCIPLINA) REFERENCES MATERIAS (CODIGO_DISCIPLINA),
CONSTRAINTS CODIGO_CURSO_5_FK FOREIGN KEY (CODIGO_CURSO) REFERENCES CURSOS (CODIGO_CURSO),
CONSTRAINTS ANO_SEMESTRE_4_FK FOREIGN KEY (ANO_SEMESTRE) REFERENCES TURMA (ANO_SEMESTRE),
CONSTRAINTS MATRICULA_ESTUDANTE_3_FK FOREIGN KEY (MATRICULA_ESTUDANTE) REFERENCES ESTUDANTE (MATRICULA_ESTUDANTE)
);
